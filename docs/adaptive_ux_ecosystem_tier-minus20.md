class UserSophisticationDetector {
  constructor() {
    this.indicators = {
      vocabulary: new VocabularyAnalyzer(),
      actions: new BehaviorTracker(),
      questions: new QuestionComplexityAnalyzer(),
      timeSpent: new EngagementTracker(),
      results: new PerformanceTracker()
    };
  }

  async detectUserLevel(userInteractions) {
    const signals = {
      // Language sophistication
      vocabulary: this.indicators.vocabulary.analyze(userInteractions.text),
      
      // Behavioral complexity
      actions: this.indicators.actions.getComplexityScore(userInteractions.clicks),
      
      // Question sophistication
      questions: this.indicators.questions.analyze(userInteractions.queries),
      
      // Learning speed
      progression: this.indicators.results.getProgressionRate(userInteractions.results)
    };

    const sophisticationScore = this.calculateOverallScore(signals);
    
    return this.mapToUserMode(sophisticationScore);
  }

  mapToUserMode(score) {
    if (score < 0.2) return 'little_explorer';
    if (score < 0.4) return 'teen_entrepreneur'; 
    if (score < 0.7) return 'adult_professional';
    if (score < 0.85) return 'senior_simplified';
    return 'quant_professional';
  }
}

class MultiPlatformRenderer {
