#### **Odds Calculation Algorithm**
```javascript
function calculateDynamicOdds(pool) {
    const totalStakes = pool.yes_stakes + pool.no_stakes + pool.house_liquidity;
    const yesProb = (pool.no_stakes + pool.house_liquidity / 2) / totalStakes;
    const noProb = (pool.yes_stakes + pool.house_liquidity / 2) / totalStakes;
    
    // Apply house edge and crowd sentiment
    const houseEdge = 0.05;
    const crowdSentiment = getCrowdSentiment(pool.arena_id);
    const sentimentModifier = 1 + (crowdSentiment.bias - 0.5) * 0.2;
    
    return {
        yes: Math.max(1.01, (1 / yesProb) * (1 - houseEdge) * sentimentModifier),
        no: Math.max(1.01, (1 / noProb) * (1 - houseEdge) / sentimentModifier)
    };
}
```
